CREATE TABLE t_board_imgs (
    id SERIAL PRIMARY KEY,
    imgurl VARCHAR NOT NULL DEFAULT '',
    original_filename VARCHAR NOT NULL DEFAULT '',
    unique_filename VARCHAR NOT NULL DEFAULT '',
    minetype VARCHAR NOT NULL DEFAULT '',
    filesize INT DEFAULT 0,
    board_id INT4 NOT NULL,
    created_dt TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_dt TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    CONSTRAINT fk_board
        FOREIGN KEY (board_id)
        REFERENCES t_board (id)
        ON DELETE CASCADE -- 옵션: t_board 레코드 삭제 시 관련 이미지도 삭제
);
